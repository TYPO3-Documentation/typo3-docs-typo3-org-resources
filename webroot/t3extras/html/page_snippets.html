lib.rootline = HMENU
lib.rootline {
	entryLevel = 0
	 
	1 = TMENU
	1 {
		noBlur=1
		wrap = <ul>
		 
		NO {
			doNotLinkIt = 1
			doNotShowLink = 1
		}
		 
		ACT = 1
		ACT {
			wrapItemAndSub = <li>|</li>
			stdWrap.field = nav_title // title
			ATagTitle.field = nav_title // title
		}
		 
		CUR = 1
		CUR {
			allWrap = <li>|</li>
			stdWrap.field = nav_title // title
			doNotLinkIt=1
		}
	}
